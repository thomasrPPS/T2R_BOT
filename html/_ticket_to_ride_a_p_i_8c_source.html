100  *
101  * Parameters:
102  * - gameType: string (max 200 characters) type of the game we want to play
103  * (empty string for regular game)
104  * "TRAINING xxxx" to play with the bot xxxx
105  * "TOURNAMENT xxxx" to join the tournament xxxx
106  * gameType can also contains extra data in form "key1=value1 key2=value1 ..."
107  * to provide options (to bots)
108  * invalid keys are ignored, invalid values leads to error
109  * the options are:
110  * - 'timeout': allows an define the timeout when training (in seconds)
111  * - 'seed': allows to set the seed of the random generator
112  * - 'start': allows to set who starts ('0' to begin, '1' otherwise)
113  * - 'map': allows to choose a map ('USA' for the moment)
114  * the following bots are available:
115  * - DO_NOTHING (stupid player what withdraw cards)
116  *
117  * - gameName: char* to get the game Name (should be allocated, max 50 characters),
118  *
119  * - nbCities: to get the number of cities
120  * - nbTracks: to get the number of tracks between the cities
121  */

136 /* ------------------------------------------------------------
137  * Get the map, the decks and initial cards and tell who starts
138  * the three arrays are filled by the function
139  *
140  * Parameters:
141  * - tracks: array of (5 x number of tracks) integers
142  * Five integers are used to define a track:
143  * - (1) id of the 1st city
144  * - (2) id of the 2nd city
145  * - (3) length of the track (between 1 and 6)
146  * - (4) color of the track (MULTICOLOR if any color can be used)
147  * - (5) color of the 2nd track if the track is double (NONE if the track is not a double track)
148  * - faceUp: array of 5 t_color giving the 5 face up cards
149  * - cards: array of 4 t_colors with the initial cards in your hand
150  *
151  * (the pointers data MUST HAVE allocated with the right size !!)
152  *
153  * Returns 0 if you begin, or 1 if the opponent begins
154  */
155 int getMap(int* tracks, t_color faceUp[5], t_color cards[4])
156 {
157  char data[4096]; /* 16 char per track, 256 tracks max */
158  int nbchar;
159  char *p, **name;
160  char city[20];
161 
162  /* check parameter */
163  if (!tracks)
164  dispError(__FUNCTION__, "The parameter `tracks` is NULL !");
165 
166  /* wait for a game */
167  int ret = getGameData( __FUNCTION__, data, 4096);
168 
169  /* copy the city's names */
170  p = data;
171  name = cityNames;
172  for(int i=0; i < nbC; i++){
173  sscanf(p, "%s%n", city, &nbchar);
174  p += nbchar;
175  *name = (char*) malloc(strlen(city)+1);
176  strCpyReplace(*(name++), city);
177  }
178 
179  /* copy the data in the tracks array */
180  for(int i=0; i < nbTr; i++){
181  sscanf(p, "%d %d %d %d %d %n", tracks, tracks+1, tracks+2, tracks+3, tracks+4, &nbchar);
182  tracks += 5;
183  p += nbchar;
184  }
185 
186  /* get the 5 face up cards */
187  sscanf(p, "%d %d %d %d %d %n", (int*)faceUp, (int*)faceUp+1, (int*)faceUp+2, (int*)faceUp+3, (int*)faceUp+4, &nbchar);
188  p += nbchar;
189  /* get the 4 initial cards */
190  sscanf(p, "%d %d %d %d", (int*)cards, (int*)cards+1, (int*)cards+2, (int*)cards+3);
191 
192  return ret;
193 }

197 /* ----------------------
198  * Get the opponent move
199  *
200  * Parameters:
201  * - move: a t_move variable, filled by the function
202  * - replay: boolean, tells if the player must replay after this move or not
203  *
204  * Returns:
205  * - NORMAL_MOVE for normal move,
206  * - WINNING_MOVE for a winning move, -1
207  * - LOOSING_MOVE for a losing (or illegal) move
208  * - this code is relative to the opponent (WINNING_MOVE if HE wins, ...)
209  */
210 t_return_code getMove(t_move* move, int* replay)
211 {
212  char moveStr[MAX_GET_MOVE];
213  char msg[MAX_MESSAGE];
214  int obj[3];
215  char* p;
216  unsigned int nbchar;
217 
218  /* get the move */
219  t_return_code ret = getCGSMove(__FUNCTION__, moveStr, msg);
220 
221  /* extract result */
222  if (ret == NORMAL_MOVE) {
223  sscanf(moveStr, "%d%n", (int*) &move->type, &nbchar);
224  p = moveStr + nbchar;
225  if (move->type == CLAIM_ROUTE) {
226  sscanf(p, "%d %d %d %d", &move->claimRoute.city1, &move->claimRoute.city2, (int*)&move->claimRoute.color, &move->claimRoute.nbLocomotives);
227  *replay = 0;
228  }
229  else if (move->type == DRAW_CARD) {
230  sscanf(msg, "%d %d %d %d %d %d %d", replay, (int*) &move->drawCard.card, (int*) move->drawCard.faceUp,
231  (int*)move->drawCard.faceUp + 1,
232  (int*)move->drawCard.faceUp + 2, (int*)move->drawCard.faceUp + 3, (int*)move->drawCard.faceUp + 4);
233  }
234  else if (move->type == DRAW_BLIND_CARD){
235  sscanf(msg, "%d", replay);
236  move->drawBlindCard.card = NONE; /* we don't know which card the opponent has */
237  }
238  else if (move->type == DRAW_OBJECTIVES) {
239  *replay = 1;
240  }
241  else if (move->type == CHOOSE_OBJECTIVES) {
242  sscanf(p, "%d %d %d", obj, obj + 1, obj + 2);
243  /* get the number of objectives kept by the opponent*/
244  move->chooseObjectives.nbObjectives = (obj[0] != 0) + (obj[1] != 0) + (obj[2] != 0);
245  *replay =0;
246  }
247  }
248 
249  return ret;
250 }
